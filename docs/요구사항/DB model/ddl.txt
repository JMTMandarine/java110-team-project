-- 회원
CREATE TABLE memb (
	mno      INTEGER      NOT NULL COMMENT '회원번호', -- 회원번호
	id       VARCHAR(50)  NOT NULL COMMENT '아이디', -- 아이디
	email    VARCHAR(40)  NOT NULL COMMENT '이메일', -- 이메일
	password VARCHAR(16)  NOT NULL COMMENT '비밀번호', -- 비밀번호
	name     VARCHAR(50)  NOT NULL COMMENT '이름', -- 이름
	nik      VARCHAR(50)  NOT NULL COMMENT '닉네임', -- 닉네임
	phot     VARCHAR(255) NULL     COMMENT '회원사진', -- 회원사진
	cdt      DATE         NOT NULL COMMENT '가입일', -- 가입일
	genre1   VARCHAR(50)  NOT NULL COMMENT '선호장르1', -- 선호장르1
	genre2   VARCHAR(50)  NOT NULL COMMENT '선호장르2', -- 선호장르2
	genre3   VARCHAR(50)  NOT NULL COMMENT '선호장르3', -- 선호장르3
	yn       CHAR(1)      NOT NULL COMMENT '정지여부', -- 정지여부
	memo     VARCHAR(50)  NULL     COMMENT '메모' -- 메모
)
COMMENT '회원';

ALTER TABLE memb
	ADD CONSTRAINT PK_memb -- 회원 기본키
		PRIMARY KEY (
			mno -- 회원번호
		);
